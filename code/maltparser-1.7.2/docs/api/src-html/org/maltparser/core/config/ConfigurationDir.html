082            /**
083             * Creates a new configuration directory or a configuration directory from a mco-file
084             * 
085             * @param name  the name of the configuration
086             * @param type  the type of configuration
087             * @param containerIndex        the container index
088             * @throws MaltChainedException
089             */
090            public ConfigurationDir(String name, String type, int containerIndex) throws MaltChainedException {
091                    setContainerIndex(containerIndex);
092    
093                    initWorkingDirectory();
094                    if (name != null && name.length() > 0 && type != null && type.length() > 0) {
095                            setName(name);
096                            setType(type);
097                    } else {
098                            throw new ConfigurationException("The configuration name is not specified. ");
099                    }
100    
101                    setConfigDirectory(new File(workingDirectory.getPath()+File.separator+getName()));
102                    
103                    String mode = OptionManager.instance().getOptionValue(containerIndex, "config", "flowchart").toString().trim();
104                    if (mode.equals("parse")) {
105                            // During parsing also search for the MaltParser configuration file in the class path
106                            File mcoPath = new File(workingDirectory.getPath()+File.separator+getName()+".mco");
107                            if (!mcoPath.exists()) {
108                                    String classpath = System.getProperty("java.class.path");
109                                    String[] items = classpath.split(System.getProperty("path.separator"));
110                                    boolean found = false;
111                                    for (String item : items) {
112                                            File candidateDir = new File(item);
113                                            if (candidateDir.exists() && candidateDir.isDirectory()) {
114                                                    File candidateConfigFile = new File(candidateDir.getPath()+File.separator+getName()+".mco");
115                                                    if (candidateConfigFile.exists()) {
116                                                            initWorkingDirectory(candidateDir.getPath());
117                                                            setConfigDirectory(new File(workingDirectory.getPath()+File.separator+getName()));
118                                                            found = true;
119                                                            break;
120                                                    }
121                                            }
122                                    }
123                                    if (found == false) {
124                                            throw new ConfigurationException("Couldn't find the MaltParser configuration file: " + getName()+".mco");
125                                    }
126                            }
127                    try {
128                            url = mcoPath.toURI().toURL();
129                    } catch (MalformedURLException e) {
130                            // should never happen
131                            throw new ConfigurationException("File path could not be represented as a URL.");
132                    }
133                    }
134            }

136            public void initDataFormat() throws MaltChainedException {
137                    String inputFormatName = OptionManager.instance().getOptionValue(containerIndex, "input", "format").toString().trim();
138                    String outputFormatName = OptionManager.instance().getOptionValue(containerIndex, "output", "format").toString().trim();
139                    final URLFinder f = new URLFinder();
140    
141                    if (configDirectory != null && configDirectory.exists()) {
142                            if (outputFormatName.length() == 0 || inputFormatName.equals(outputFormatName)) {
143                                    URL inputFormatURL = f.findURLinJars(inputFormatName);
144                                    if (inputFormatURL != null) {
145                                            outputFormatName = inputFormatName = this.copyToConfig(inputFormatURL);
146                                    } else {
147                                            outputFormatName = inputFormatName = this.copyToConfig(inputFormatName);
148                                    }
149                            } else {
150                                    URL inputFormatURL = f.findURLinJars(inputFormatName);
151                                    if (inputFormatURL != null) {
152                                            inputFormatName = this.copyToConfig(inputFormatURL);
153                                    } else {
154                                            inputFormatName = this.copyToConfig(inputFormatName);
155                                    }
156                                    URL outputFormatURL = f.findURLinJars(outputFormatName);
157                                    if (inputFormatURL != null) {
158                                            outputFormatName = this.copyToConfig(outputFormatURL);
159                                    } else {
160                                            outputFormatName = this.copyToConfig(outputFormatName);
161                                    }
162                            }
163                            OptionManager.instance().overloadOptionValue(containerIndex, "input", "format", inputFormatName);
164                    } else {
165                            if (outputFormatName.length() == 0) {
166                                    outputFormatName = inputFormatName;
167                            }
168                    }
169                    dataFormatInstances = new HashMap<String, DataFormatInstance>(3);
170                    inputFormatURL = findURL(inputFormatName);
171                    outputFormatURL = findURL(outputFormatName);
172                    if (outputFormatURL != null) {
173                            try {
174                                    InputStream is = outputFormatURL.openStream();
175                            } catch (FileNotFoundException e) {
176                                    outputFormatURL = f.findURL(outputFormatName);
177                            } catch (IOException e) {
178                                    outputFormatURL = f.findURL(outputFormatName);
179                            }
180                    } else {
181                            outputFormatURL = f.findURL(outputFormatName);
182                    }
183                    dataFormatManager = new DataFormatManager(inputFormatURL, outputFormatURL);
184                    
185                    String mode = OptionManager.instance().getOptionValue(containerIndex, "config", "flowchart").toString().trim();
186                    if (mode.equals("parse")) {
187                            symbolTables = new TrieSymbolTableHandler(TrieSymbolTableHandler.ADD_NEW_TO_TMP_STORAGE);
188    //                      symbolTables = new TrieSymbolTableHandler(TrieSymbolTableHandler.ADD_NEW_TO_TRIE);
189                    } else {
190                            symbolTables = new TrieSymbolTableHandler(TrieSymbolTableHandler.ADD_NEW_TO_TRIE);
191                    }
192                    if (dataFormatManager.getInputDataFormatSpec().getDataStructure() == DataStructure.PHRASE) {
193                            if (mode.equals("learn")) {
194                                    Set<Dependency> deps = dataFormatManager.getInputDataFormatSpec().getDependencies();
195                                    for (Dependency dep : deps) {
196                                            URL depFormatURL = f.findURLinJars(dep.getUrlString());
197                                            if (depFormatURL != null) {
198                                                    this.copyToConfig(depFormatURL);
199                                            } else {
200                                                    this.copyToConfig(dep.getUrlString());
201                                            }
202                                    }
203                            } 
204                            else if (mode.equals("parse")) {
205                                    Set<Dependency> deps = dataFormatManager.getInputDataFormatSpec().getDependencies();
206                                    String nullValueStategy = OptionManager.instance().getOptionValue(containerIndex, "singlemalt", "null_value").toString();
207                                    for (Dependency dep : deps) {
208    //                                      URL depFormatURL = f.findURLinJars(dep.getUrlString());
209                                            DataFormatInstance dataFormatInstance = dataFormatManager.getDataFormatSpec(dep.getDependentOn()).createDataFormatInstance(symbolTables, nullValueStategy);
210                                            addDataFormatInstance(dataFormatManager.getDataFormatSpec(dep.getDependentOn()).getDataFormatName(), dataFormatInstance);
211                                            dataFormatManager.setInputDataFormatSpec(dataFormatManager.getDataFormatSpec(dep.getDependentOn()));
212    //                                      dataFormatManager.setOutputDataFormatSpec(dataFormatManager.getDataFormatSpec(dep.getDependentOn()));
213                                    }
214                            }
215                    }
216            }

218            private URL findURL(String specModelFileName) throws MaltChainedException {
219                    URL url = null;
220                    File specFile = this.getFile(specModelFileName);
221                    if (specFile.exists()) {
222                            try {
223                                    url = new URL("file:///"+specFile.getAbsolutePath());
224                            } catch (MalformedURLException e) {
225                                    throw new MaltChainedException("Malformed URL: "+specFile, e);
226                            }
227                    } else {
228                            url = this.getConfigFileEntryURL(specModelFileName);
229                    }
230                    return url;
231            }

467        /**
468         * Copies a file into the configuration directory.
469         * 
470         * @param source    a path to file 
471         * @throws MaltChainedException
472         */
473        public String copyToConfig(File source) throws MaltChainedException {
474            byte[] readBuffer = new byte[BUFFER];
475            String destination = configDirectory.getPath()+File.separator+source.getName();
476            try {
477                    BufferedInputStream bis = new BufferedInputStream(new FileInputStream(source));
478                    BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(destination), BUFFER);
479                
480                    int n = 0;
481                        while ((n = bis.read(readBuffer, 0, BUFFER)) != -1) {
482                            bos.write(readBuffer, 0, n);
483                        }
484                    bos.flush();
485                    bos.close();
486                    bis.close();
487                    } catch (FileNotFoundException e) {
488                            throw new ConfigurationException("The source file '"+source+"' cannot be found or the destination file '"+destination+"' cannot be created when coping the file. ", e);
489                    } catch (IOException e) {
490                            throw new ConfigurationException("The source file '"+source+"' cannot be copied to destination '"+destination+"'. ", e);
491                    }
492                    return source.getName();
493        }

496        public String copyToConfig(String fileUrl) throws MaltChainedException {
497            final URLFinder f = new URLFinder();
498            URL url = f.findURL(fileUrl);
499            if (url == null) {
500                    throw new ConfigurationException("The file or URL '"+fileUrl+"' could not be found. ");
501            }
502            return copyToConfig(url);
503        }

505        public String copyToConfig(URL url) throws MaltChainedException {
506            if (url == null) {
507                    throw new ConfigurationException("URL could not be found. ");
508            }
509            byte[] readBuffer = new byte[BUFFER];
510            String destFileName = url.getPath();
511                    int indexSlash = destFileName.lastIndexOf('/');
512                    if (indexSlash == -1) {
513                            indexSlash = destFileName.lastIndexOf('\\');
514                    }
515            
516                    if (indexSlash != -1) {
517                            destFileName = destFileName.substring(indexSlash+1);
518                    }
519            
520            String destination = configDirectory.getPath()+File.separator+destFileName;
521            try {
522                    BufferedInputStream bis = new BufferedInputStream(url.openStream());
523                    BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(destination), BUFFER);
524                
525                    int n = 0;
526                        while ((n = bis.read(readBuffer, 0, BUFFER)) != -1) {
527                            bos.write(readBuffer, 0, n);
528                        }
529                    bos.flush();
530                    bos.close();
531                    bis.close();
532                    } catch (FileNotFoundException e) {
533                            throw new ConfigurationException("The destination file '"+destination+"' cannot be created when coping the file. ", e);
534                    } catch (IOException e) {
535                            throw new ConfigurationException("The URL '"+url+"' cannot be copied to destination '"+destination+"'. ", e);
536                    }
537                    return destFileName;
538        }

541            /**
542             * Removes the configuration directory, if it exists and it contains a .info file. 
543             * 
544             * @throws MaltChainedException
545             */
546            public void deleteConfigDirectory() throws MaltChainedException {
547                    if (!configDirectory.exists()) {
548                            return;
549                    }
550                    File infoFile = new File(configDirectory.getPath()+File.separator+getName()+"_"+getType()+".info");
551                    if (infoFile.exists()) {
552                            deleteConfigDirectory(configDirectory);
553                    } else {
554                            throw new ConfigurationException("There exists a directory that is not a MaltParser configuration directory. ");
555                    }
556            }
557            
558            private void deleteConfigDirectory(File directory) throws MaltChainedException {
559                    if (directory.exists()) {
560                            File[] files = directory.listFiles();
561                            for (int i = 0; i < files.length; i++) {
562                                    if (files[i].isDirectory()) {
563                                            deleteConfigDirectory(files[i]);
564                                    } else {
565                                            files[i].delete();
566                                    }
567                            }
568                    } else {
569                            throw new ConfigurationException("The directory '"+directory.getPath()+ "' cannot be found. ");
570                    }
571                    directory.delete();
572            }

663            private void createConfigFile(String directory, JarOutputStream jos) throws MaltChainedException {
664            byte[] readBuffer = new byte[BUFFER];
665                    try {
666                            File zipDir = new File(directory);
667                            String[] dirList = zipDir.list();
668                            
669                            int bytesIn = 0;
670            
671                            for (int i = 0; i < dirList.length; i++) {
672                                    File f = new File(zipDir, dirList[i]);
673                                    if (f.isDirectory()) {
674                                            String filePath = f.getPath();
675                                            createConfigFile(filePath, jos);
676                                            continue;
677                                    }
678            
679                                    FileInputStream fis = new FileInputStream(f);
680                                    
681                                    String entryPath = f.getPath().substring(workingDirectory.getPath().length()+1);
682                                    entryPath = entryPath.replace('\\', '/');
683                                    JarEntry entry = new JarEntry(entryPath);
684                                    jos.putNextEntry(entry);
685            
686                                    while ((bytesIn = fis.read(readBuffer)) != -1) {
687                                            jos.write(readBuffer, 0, bytesIn);
688                                    }
689            
690                                    fis.close();
691                            }
692                    } catch (FileNotFoundException e) {
693                            throw new ConfigurationException("The directory '"+directory+"' cannot be found. ", e);
694                    } catch (IOException e) {
695                            throw new ConfigurationException("The directory '"+directory+"' cannot be compressed into a mco file. ", e);
696                    } 
697            }

700            public void copyConfigFile(File in, File out, Versioning versioning) throws MaltChainedException {
701                    try {
702                            JarFile jar = new JarFile(in);
703                            JarOutputStream tempJar = new JarOutputStream(new FileOutputStream(out));
704                    byte[] buffer = new byte[BUFFER];
705                    int bytesRead;
706                    final StringBuilder sb = new StringBuilder();
707                    final URLFinder f = new URLFinder();
708    
709                    for (Enumeration<JarEntry> entries = jar.entries(); entries.hasMoreElements(); ) {
710                        JarEntry inEntry = (JarEntry) entries.nextElement();
711                        InputStream entryStream = jar.getInputStream(inEntry);
712                        JarEntry outEntry = versioning.getJarEntry(inEntry);
713                        
714                        if (!versioning.hasChanges(inEntry, outEntry)) {
715                                tempJar.putNextEntry(outEntry);
716                                while ((bytesRead = entryStream.read(buffer)) != -1) {
717                                    tempJar.write(buffer, 0, bytesRead);
718                                }
719                        } else {
720                            tempJar.putNextEntry(outEntry);
721                            BufferedReader br = new BufferedReader(new InputStreamReader(entryStream));
722                            String line = null;
723                            sb.setLength(0);
724                            while ((line = br.readLine()) != null) {
725                                    sb.append(line);
726                                    sb.append('\n');
727                            }
728                            String outString = versioning.modifyJarEntry(inEntry, outEntry, sb);
729                            tempJar.write(outString.getBytes());
730                        }
731                    }
732                    if (versioning.getFeatureModelXML() != null && versioning.getFeatureModelXML().startsWith("/appdata")) {
733                            int index = versioning.getFeatureModelXML().lastIndexOf('/');
734                    BufferedInputStream bis = new BufferedInputStream(f.findURLinJars(versioning.getFeatureModelXML()).openStream());
735                    tempJar.putNextEntry(new JarEntry(versioning.getNewConfigName()+"/" +versioning.getFeatureModelXML().substring(index+1)));
736                    int n = 0;
737                        while ((n = bis.read(buffer, 0, BUFFER)) != -1) {
738                            tempJar.write(buffer, 0, n);
739                        }
740                    bis.close();
741                    }
742                    if (versioning.getInputFormatXML() != null && versioning.getInputFormatXML().startsWith("/appdata")) {
743                            int index = versioning.getInputFormatXML().lastIndexOf('/');
744                    BufferedInputStream bis = new BufferedInputStream(f.findURLinJars(versioning.getInputFormatXML()).openStream());
745                    tempJar.putNextEntry(new JarEntry(versioning.getNewConfigName()+"/" +versioning.getInputFormatXML().substring(index+1)));
746                    int n = 0;
747                        while ((n = bis.read(buffer, 0, BUFFER)) != -1) {
748                            tempJar.write(buffer, 0, n);
749                        }
750                    bis.close();
751                    }
752                    tempJar.flush();
753                    tempJar.close();
754                    jar.close();
755                    } catch (IOException e) {
756                            throw new ConfigurationException("", e);
757                    }
758            }

761                    if (url == null) {
762                            throw new ConfigurationException("The URL cannot be found. ");
763                    }       
764                    try {
765                            JarEntry je;
766                            JarInputStream jis = new JarInputStream(url.openConnection().getInputStream());
767                            while ((je = jis.getNextJarEntry()) != null) {
768                                    String entryName = je.getName();
769                                    if (entryName.endsWith(".info")) {
770                                            int indexUnderScore = entryName.lastIndexOf('_');
771                                            int indexSeparator = entryName.lastIndexOf(File.separator);
772                                            if (indexSeparator == -1) {
773                                                    indexSeparator = entryName.lastIndexOf('/');
774                                            }
775                                            if (indexSeparator == -1) {
776                                                    indexSeparator = entryName.lastIndexOf('\\');
777                                            }
778                                            int indexDot = entryName.lastIndexOf('.');
779                                            if (indexUnderScore == -1 || indexDot == -1) {
780                                                    throw new ConfigurationException("Could not find the configuration name and type from the URL '"+url.toString()+"'. ");
781                                            }
782                                            setName(entryName.substring(indexSeparator+1, indexUnderScore));
783                                            setType(entryName.substring(indexUnderScore+1, indexDot));
784                                            setConfigDirectory(new File(workingDirectory.getPath()+File.separator+getName()));
785                                            jis.close();
786                                            return;
787                                    }
788                            }
789                            
790                    } catch (IOException e) {
791                            throw new ConfigurationException("Could not find the configuration name and type from the URL '"+url.toString()+"'. ", e);
792                    }
793        }

857        protected void unpackConfigFile(JarInputStream jis) throws MaltChainedException {
858                    try {
859                            JarEntry je;
860                            byte[] readBuffer = new byte[BUFFER];
861                    SortedSet<String> directoryCache  = new TreeSet<String>();
862                            while ((je = jis.getNextJarEntry()) != null) {
863                            String entryName = je.getName();
864    
865                            if (entryName.startsWith("/")) {
866                                    entryName = entryName.substring(1);
867                            }
868                            if (entryName.endsWith(File.separator) || entryName.endsWith("/")) {
869                                return;
870                            }
871                            int index = -1;
872                            if (File.separator.equals("\\")) {
873                                    entryName = entryName.replace('/', '\\');
874                                    index = entryName.lastIndexOf("\\");
875                            } else if (File.separator.equals("/")) {
876                                    entryName = entryName.replace('\\', '/');
877                                    index = entryName.lastIndexOf("/");
878                            }
879                            if (index > 0) {
880                                String dirName = entryName.substring(0, index);
881                                if (!directoryCache.contains(dirName)) {
882                                    File directory = new File(workingDirectory.getPath()+File.separator+dirName);
883                                    if (!(directory.exists() && directory.isDirectory())) {
884                                        if (!directory.mkdirs()) {
885                                            throw new ConfigurationException("Unable to make directory '" + dirName +"'. ");
886                                        }
887                                        directoryCache.add(dirName);
888                                    }
889                                }
890                            }
891                           
892                            if (new File(workingDirectory.getPath()+File.separator+entryName).isDirectory() && new File(workingDirectory.getPath()+File.separator+entryName).exists()) {
893                                    continue;
894                            }
895                            BufferedOutputStream bos;
896                            try {
897                                    bos = new BufferedOutputStream(new FileOutputStream(workingDirectory.getPath()+File.separator+entryName), BUFFER);
898                            } catch (FileNotFoundException e) {
899                                            throw new ConfigurationException("Could not unpack configuration. The file '"+workingDirectory.getPath()+File.separator+entryName+"' cannot be unpacked. ", e);
900                            }
901                                    int n = 0;
902                                while ((n = jis.read(readBuffer, 0, BUFFER)) != -1) {
903                                    bos.write(readBuffer, 0, n);
904                                }
905                                bos.flush();
906                                bos.close();
907                            }
908                    } catch (IOException e) {
909                            throw new ConfigurationException("Could not unpack configuration. ", e);
910                    }
911        }

1048                    initCreatedByMaltParserVersionFromInfoFile();
1049                    SystemLogger.logger().info("\nCurrent version      : " + SystemInfo.getVersion() + "\n");
1050                    SystemLogger.logger().info("Parser model version : " + createdByMaltParserVersion + "\n");
1051                    if (SystemInfo.getVersion() == null) {
1052                            throw new ConfigurationException("Couln't determine the version of MaltParser");
1053                    } else if (createdByMaltParserVersion == null) {
1054                            throw new ConfigurationException("Couln't determine the version of the parser model");
1055                    } else if (SystemInfo.getVersion().equals(createdByMaltParserVersion)) {
1056                            SystemLogger.logger().info("The parser model "+getName()+".mco has already the same version as the current version of MaltParser. \n");

1063                    if (!versioning.support(createdByMaltParserVersion)) {
1064                            SystemLogger.logger().warn("The parser model '"+ name+ ".mco' is created by MaltParser "+getCreatedByMaltParserVersion()+", which cannot be converted to a MaltParser "+SystemInfo.getVersion()+" parser model.\n");
1065                            SystemLogger.logger().warn("Please retrain the parser model with MaltParser "+SystemInfo.getVersion() +" or download MaltParser "+getCreatedByMaltParserVersion()+" from http://maltparser.org/download.html\n");

1068                    SystemLogger.logger().info("Converts the parser model '"+ mcoPath.getName()+ "' into '"+newMcoPath.getName()+"'....\n");
1069                    copyConfigFile(mcoPath, newMcoPath, versioning);
1070            }
