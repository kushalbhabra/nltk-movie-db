048            protected void trainInternal(FeatureVector featureVector) throws MaltChainedException {
049                    if (configLogger.isInfoEnabled()) {
050                            configLogger.info("Creating Liblinear model "+getFile(".moo").getName()+"\n");
051                    }
052                    double[] wmodel = null;
053                    int[] labels = null;
054                    int nr_class = 0;
055                    int nr_feature = 0;
056                    Parameter parameter = getLiblinearParameters();
057                    try {   
058                            Problem problem = readProblem(getInstanceInputStreamReader(".ins"));
059                            boolean res = checkProblem(problem);
060                            if (res == false) {
061                                    throw new LibException("Abort (The number of training instances * the number of classes) > "+Integer.MAX_VALUE+" and this is not supported by LibLinear. ");
062                            }
063                            if (configLogger.isInfoEnabled()) {
064                                    owner.getGuide().getConfiguration().getConfigLogger().info("- Train a parser model using LibLinear.\n");
065                            }
066                            final PrintStream out = System.out;
067                            final PrintStream err = System.err;
068                            System.setOut(NoPrintStream.NO_PRINTSTREAM);
069                            System.setErr(NoPrintStream.NO_PRINTSTREAM);
070                            Model model = Linear.train(problem, parameter);
071                            System.setOut(err);
072                            System.setOut(out);
073                            problem = null;
074                            wmodel = model.getFeatureWeights();
075                            labels = model.getLabels();
076                            nr_class = model.getNrClass();
077                            nr_feature = model.getNrFeature();
078                            if (!saveInstanceFiles) {
079                                    getFile(".ins").delete();
080                            }
081                    } catch (OutOfMemoryError e) {
082                            throw new LibException("Out of memory. Please increase the Java heap size (-Xmx<size>). ", e);
083                    } catch (IllegalArgumentException e) {
084                            throw new LibException("The Liblinear learner was not able to redirect Standard Error stream. ", e);
085                    } catch (SecurityException e) {
086                            throw new LibException("The Liblinear learner cannot remove the instance file. ", e);
087                    } catch (NegativeArraySizeException e) {
088                            throw new LibException("(The number of training instances * the number of classes) > "+Integer.MAX_VALUE+" and this is not supported by LibLinear.", e);
089                    }
090                    
091                    if (configLogger.isInfoEnabled()) {
092                            configLogger.info("- Optimize the memory usage\n");
093                    }
094                    MaltLiblinearModel xmodel = null;
095                    try {
096    //                      System.out.println("Nr Features:" +  nr_feature);
097    //                      System.out.println("nr_class:" + nr_class);
098    //                      System.out.println("wmodel.length:" + wmodel.length);           
099                            double[][] wmatrix = convert2(wmodel, nr_class, nr_feature);
100                            xmodel = new MaltLiblinearModel(labels, nr_class, wmatrix.length, wmatrix, parameter.getSolverType());
101                            if (configLogger.isInfoEnabled()) {
102                                    configLogger.info("- Save the Liblinear model "+getFile(".moo").getName()+"\n");
103                            }
104                    } catch (OutOfMemoryError e) {
105                            throw new LibException("Out of memory. Please increase the Java heap size (-Xmx<size>). ", e);
106                    }                       
107                    try {
108                            if (xmodel != null) {
109                                ObjectOutputStream output = new ObjectOutputStream (new BufferedOutputStream(new FileOutputStream(getFile(".moo").getAbsolutePath())));
110                            try{
111                              output.writeObject(xmodel);
112                            } finally {
113                              output.close();
114                            }
115                            }
116                    } catch (OutOfMemoryError e) {
117                            throw new LibException("Out of memory. Please increase the Java heap size (-Xmx<size>). ", e);
118                    } catch (IllegalArgumentException e) {
119                            throw new LibException("The Liblinear learner was not able to redirect Standard Error stream. ", e);
120                    } catch (SecurityException e) {
121                            throw new LibException("The Liblinear learner cannot remove the instance file. ", e);
122                    } catch (IOException e) {
123                            throw new LibException("The Liblinear learner cannot save the model file '"+getFile(".mod").getAbsolutePath()+"'. ", e);
124                    }
125            }

127        private double[][] convert2(double[] w, int nr_class, int nr_feature) {
128            int[] wlength = new int[nr_feature];
129            int nr_nfeature = 0;
130    //        int ne = 0;
131    //        int nr = 0;
132    //        int no = 0;
133    //        int n = 0;
134            
135            // Identify length of new weight array for each feature
136            for (int i = 0; i < nr_feature; i++) {
137                    int k = nr_class;               
138                    for (int t = i * nr_class; (t + (k - 1)) >= t; k--) {
139                            if (w[t + k - 1] != 0.0) {
140                                    break;
141                            }
142                    }
143                    int b = k;
144                    if (b != 0) {
145                            for (int t = i * nr_class; (t + (b - 1)) >= t; b--) {
146                                    if (b != k) {
147                                            if (w[t + b - 1] != w[t + b]) {
148                                                    break;
149                                            }
150                                    }
151                            }
152                    }
153                    if (k == 0 || b == 0) {
154                            wlength[i] = 0;
155                    } else {
156                            wlength[i] = k;
157                            nr_nfeature++;
158                    }               
159            }
160            // Allocate the weight matrix with the new number of features and
161            // an array wsignature that efficient compare if weight vector can be reused by another feature. 
162            double[][] wmatrix = new double[nr_nfeature][];
163            double[] wsignature = new double[nr_nfeature];
164            Long[] reverseMap = featureMap.reverseMap();
165            int in = 0;
166            for (int i = 0; i < nr_feature; i++) {
167                if (wlength[i] == 0) {
168                    // if the length of the weight vector is zero than eliminate the feature from the feature map.
169    //              ne++;
170                    featureMap.removeIndex(reverseMap[i + 1]);
171                    reverseMap[i + 1] = null;
172                } else {            
173                    boolean reuse = false;
174                    double[] copy = new double[wlength[i]];
175                    System.arraycopy(w, i * nr_class, copy, 0, wlength[i]);
176                    featureMap.setIndex(reverseMap[i + 1], in + 1);
177                    for (int j=0; j<copy.length; j++) wsignature[in] += copy[j];
178                        for (int j = 0; j < in; j++) {
179                            if (wsignature[j] == wsignature[in]) {
180                                    // if the signatures is equal then do more narrow comparison  
181                                    if (Util.equals(copy, wmatrix[j])) {
182                                            // if equal then reuse the weight vector
183                                            wmatrix[in] = wmatrix[j];
184                                            reuse = true;
185    //                                      nr++;
186                                            break;
187                                    }
188                            }
189                        }
190                        if (reuse == false) {
191                            // if no reuse has done use the new weight vector in the weight matrix 
192    //                      no++;
193                            wmatrix[in] = copy;
194                        }
195                        in++;
196                }
197    //            n++;
198            }
199            featureMap.setFeatureCounter(nr_nfeature);
200    //        System.out.println("NE:"+ne);
201    //        System.out.println("NR:"+nr);
202    //        System.out.println("NO:"+no);
203    //        System.out.println("N :"+n);
204            return wmatrix;
205        }

207        private double[][] convert(double[] w, int nr_class, int nr_feature) {
208            double[][] wmatrix = new double[nr_feature][];
209            double[] wsignature = new double[nr_feature];
210            boolean reuse = false;
211            int ne = 0;
212            int nr = 0;
213            int no = 0;
214            int n = 0;
215            Long[] reverseMap = featureMap.reverseMap();
216            for (int i = 0; i < nr_feature; i++) {
217                    reuse = false;
218                    int k = nr_class;
219                    for (int t = i * nr_class; (t + (k - 1)) >= t; k--) {
220                            if (w[t + k - 1] != 0.0) {
221                                    break;
222                            }
223                    }
224                    double[] copy = new double[k];
225                System.arraycopy(w, i * nr_class, copy, 0,k);
226                if (eliminate(copy)) {
227                    ne++;
228                    featureMap.removeIndex(reverseMap[i + 1]);
229                    reverseMap[i + 1] = null;
230                    wmatrix[i] = null;
231                } else {
232                    featureMap.setIndex(reverseMap[i + 1], i + 1 - ne);
233                    for (int j=0; j<copy.length; j++) wsignature[i] += copy[j];
234                        for (int j = 0; j < i; j++) {
235                            if (wsignature[j] == wsignature[i]) {
236                                    if (Util.equals(copy, wmatrix[j])) {
237                                            wmatrix[i] = wmatrix[j];
238                                            reuse = true;
239                                            nr++;
240                                            break;
241                                    }
242                            }
243                        }
244                        if (reuse == false) {
245                            no++;
246                            wmatrix[i] = copy;
247                        }
248                }
249                n++;
250            }
251            featureMap.setFeatureCounter(featureMap.getFeatureCounter()- ne);
252            double[][] wmatrix_reduced = new double[nr_feature-ne][];
253            for (int i = 0, j = 0; i < wmatrix.length; i++) {
254                    if (wmatrix[i] != null) {
255                            wmatrix_reduced[j++] = wmatrix[i];
256                    }
257            }
258    //        System.out.println("NE:"+ne);
259    //        System.out.println("NR:"+nr);
260    //        System.out.println("NO:"+no);
261    //        System.out.println("N :"+n);
262    
263            return wmatrix_reduced;
264        }

266        public static boolean eliminate(double[] a) {
267            if (a.length == 0) {
268                    return true;
269            }
270            for (int i = 1; i < a.length; i++) {
271                    if (a[i] != a[i-1]) {
272                            return false;
273                    }
274            }
275            return true;
276        }

278            protected void trainExternal(FeatureVector featureVector) throws MaltChainedException {
279                    try {           
280                            
281                            if (configLogger.isInfoEnabled()) {
282                                    owner.getGuide().getConfiguration().getConfigLogger().info("Creating liblinear model (external) "+getFile(".mod").getName());
283                            }
284                            binariesInstances2SVMFileFormat(getInstanceInputStreamReader(".ins"), getInstanceOutputStreamWriter(".ins.tmp"));
285                            final String[] params = getLibParamStringArray();
286                            String[] arrayCommands = new String[params.length+3];
287                            int i = 0;
288                            arrayCommands[i++] = pathExternalTrain;
289                            for (; i <= params.length; i++) {
290                                    arrayCommands[i] = params[i-1];
291                            }
292                            arrayCommands[i++] = getFile(".ins.tmp").getAbsolutePath();
293                            arrayCommands[i++] = getFile(".mod").getAbsolutePath();
294                            
295                    if (verbosity == Verbostity.ALL) {
296                            owner.getGuide().getConfiguration().getConfigLogger().info('\n');
297                    }
298                            final Process child = Runtime.getRuntime().exec(arrayCommands);
299                    final InputStream in = child.getInputStream();
300                    final InputStream err = child.getErrorStream();
301                    int c;
302                    while ((c = in.read()) != -1){
303                            if (verbosity == Verbostity.ALL) {
304                                    owner.getGuide().getConfiguration().getConfigLogger().info((char)c);
305                            }
306                    }
307                    while ((c = err.read()) != -1){
308                            if (verbosity == Verbostity.ALL || verbosity == Verbostity.ERROR) {
309                                    owner.getGuide().getConfiguration().getConfigLogger().info((char)c);
310                            }
311                    }
312                if (child.waitFor() != 0) {
313                    owner.getGuide().getConfiguration().getConfigLogger().info(" FAILED ("+child.exitValue()+")");
314                }
315                    in.close();
316                    err.close();
317                            if (configLogger.isInfoEnabled()) {
318                                    configLogger.info("\nSaving Liblinear model "+getFile(".moo").getName()+"\n");
319                            }
320                            MaltLiblinearModel xmodel = new MaltLiblinearModel(getFile(".mod"));
321                        ObjectOutputStream output = new ObjectOutputStream (new BufferedOutputStream(new FileOutputStream(getFile(".moo").getAbsolutePath())));
322                    try{
323                      output.writeObject(xmodel);
324                    } finally {
325                      output.close();
326                    }
327                    if (!saveInstanceFiles) {
328                                    getFile(".ins").delete();
329                                    getFile(".mod").delete();
330                                    getFile(".ins.tmp").delete();
331                    }
332                    if (configLogger.isInfoEnabled()) {
333                            configLogger.info('\n');
334                    }
335                    } catch (InterruptedException e) {
336                             throw new LibException("Learner is interrupted. ", e);
337                    } catch (IllegalArgumentException e) {
338                            throw new LibException("The learner was not able to redirect Standard Error stream. ", e);
339                    } catch (SecurityException e) {
340                            throw new LibException("The learner cannot remove the instance file. ", e);
341                    } catch (IOException e) {
342                            throw new LibException("The learner cannot save the model file '"+getFile(".mod").getAbsolutePath()+"'. ", e);
343                    } catch (OutOfMemoryError e) {
344                            throw new LibException("Out of memory. Please increase the Java heap size (-Xmx<size>). ", e);
345                    }
346            }

364            private Problem readProblem(InputStreamReader isr) throws MaltChainedException {
365                    Problem problem = new Problem();
366                    final FeatureList featureList = new FeatureList();
367                    if (configLogger.isInfoEnabled()) {
368                            owner.getGuide().getConfiguration().getConfigLogger().info("- Read all training instances.\n");
369                    }
370                    try {
371                            final BufferedReader fp = new BufferedReader(isr);
372                            
373                            problem.bias = -1;
374                            problem.l = getNumberOfInstances();
375                            problem.x = new FeatureNode[problem.l][];
376                            problem.y = new int[problem.l];
377                            int i = 0;
378                            
379                            while(true) {
380                                    String line = fp.readLine();
381                                    if(line == null) break;
382                                    int y = binariesInstance(line, featureList);
383                                    if (y == -1) {
384                                            continue;
385                                    }
386                                    try {
387                                            problem.y[i] = y;
388                                            problem.x[i] = new FeatureNode[featureList.size()];
389                                            int p = 0;
390                                    for (int k=0; k < featureList.size(); k++) {
391                                            MaltFeatureNode x = featureList.get(k);
392                                                    problem.x[i][p++] = new FeatureNode(x.getIndex(), x.getValue());
393                                            }
394                                            i++;
395                                    } catch (ArrayIndexOutOfBoundsException e) {
396                                            throw new LibException("Couldn't read liblinear problem from the instance file. ", e);
397                                    }
398    
399                            }
400                            fp.close();
401                            problem.n = featureMap.size();
402                    } catch (IOException e) {
403                            throw new LibException("Cannot read from the instance file. ", e);
404                    }
405                    
406                    return problem;
407            }

409            private boolean checkProblem(Problem problem) throws MaltChainedException {
410                    int max_y = problem.y[0];
411                    for (int i = 1; i < problem.y.length; i++) {
412                            if (problem.y[i] > max_y) {
413                                    max_y = problem.y[i];
414                            }
415                    }
416                    if (max_y * problem.l < 0) { // max_y * problem.l > Integer.MAX_VALUE
417                            if (configLogger.isInfoEnabled()) {
418                                    owner.getGuide().getConfiguration().getConfigLogger().info("*** Abort (The number of training instances * the number of classes) > Max array size: ("+problem.l+" * "+max_y+") > "+Integer.MAX_VALUE+" and this is not supported by LibLinear.\n");
419                            }
420                            return false;
421                    }
422                    return true;
423            }

425            private Parameter getLiblinearParameters() throws MaltChainedException {
426                    Parameter param = new Parameter(SolverType.MCSVM_CS, 0.1, 0.1);
427                    String type = libOptions.get("s");
428                    
429                    if (type.equals("0")) {
430                            param.setSolverType(SolverType.L2R_LR);
431                    } else if (type.equals("1")) {
432                            param.setSolverType(SolverType.L2R_L2LOSS_SVC_DUAL);
433                    } else if (type.equals("2")) {
434                            param.setSolverType(SolverType.L2R_L2LOSS_SVC);
435                    } else if (type.equals("3")) {
436                            param.setSolverType(SolverType.L2R_L1LOSS_SVC_DUAL);
437                    } else if (type.equals("4")) {
438                            param.setSolverType(SolverType.MCSVM_CS);
439                    } else if (type.equals("5")) {
440                            param.setSolverType(SolverType.L1R_L2LOSS_SVC); 
441                    } else if (type.equals("6")) {
442                            param.setSolverType(SolverType.L1R_LR); 
443                    } else if (type.equals("7")) {
444                            param.setSolverType(SolverType.L2R_LR_DUAL);    
445                    } else {
446                            throw new LibException("The liblinear type (-s) is not an integer value between 0 and 4. ");
447                    }
448                    try {
449                            param.setC(Double.valueOf(libOptions.get("c")).doubleValue());
450                    } catch (NumberFormatException e) {
451                            throw new LibException("The liblinear cost (-c) value is not numerical value. ", e);
452                    }
453                    try {
454                            param.setEps(Double.valueOf(libOptions.get("e")).doubleValue());
455                    } catch (NumberFormatException e) {
456                            throw new LibException("The liblinear epsilon (-e) value is not numerical value. ", e);
457                    }
458                    return param;
459            }
